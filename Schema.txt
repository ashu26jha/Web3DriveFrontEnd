type activeFile @entity {
  id: ID!
  tokenId: BigInt!
  Owner: Bytes!
  ipfsHash: String
}

type accessLevel @entity(immutable: true) {
  id: ID!
  gotAccess: Bytes! # address
  token: BigInt! # uint256
  AdminPrivilege: Int! # uint8
}

type fileAdded @entity(immutable: true) {
  id: ID!
  tokenId: BigInt! # uint256
  ipfsHash: String! # string
  Owner: Bytes! # address
}

type fileDeleted @entity(immutable: true) {
  id: ID!
  whoDeleted: Bytes! # address
  token: BigInt! # uint256
}
